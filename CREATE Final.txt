def add_customer(connection, first_name, last_name, address, phone_number, email):
    cursor = connection.cursor()
    cursor.execute('''
        INSERT INTO Customers (FirstName, LastName, Address, PhoneNumber, Email)
        VALUES (?, ?, ?, ?, ?)
    ''', (first_name, last_name, address, phone_number, email))
    connection.commit()
    print("Customer added successfully.")

def add_loan(connection, customer_id, loan_amount, loan_term, interest_rate, status="Active"):
    cursor = connection.cursor()
    cursor.execute('''
        INSERT INTO Loans (CustomerID, LoanAmount, LoanTerm, InterestRate, Status)
        VALUES (?, ?, ?, ?, ?)
    ''', (customer_id, loan_amount, loan_term, interest_rate, status))
    connection.commit()
    print("Loan added successfully.")
