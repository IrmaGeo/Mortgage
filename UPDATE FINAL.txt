def update_customer(connection, customer_id, phone_number=None, address=None):
    cursor = connection.cursor()
    if phone_number:
        cursor.execute('UPDATE Customers SET PhoneNumber = ? WHERE CustomerID = ?', (phone_number, customer_id))
    if address:
        cursor.execute('UPDATE Customers SET Address = ? WHERE CustomerID = ?', (address, customer_id))
    connection.commit()
    print("Customer updated successfully.")

def update_loan(connection, loan_id, status):
    cursor = connection.cursor()
    cursor.execute('UPDATE Loans SET Status = ? WHERE LoanID = ?', (status, loan_id))
    connection.commit()
    print("Loan updated successfully.")
